DAI_is_historical_axis_license_bloc_member = {
	OR = {
		original_tag = GER
		original_tag = ITA
		original_tag = HUN
		original_tag = ROM
		original_tag = BUL
		AND = {
			is_subject = yes
			overlord = {
				OR = {
					original_tag = GER
					original_tag = ITA
					original_tag = HUN
					original_tag = ROM
					original_tag = BUL
				}
			}
		}
	}
}

		AND = {

DAI_is_historical_license_partner_of_root = {
	is_historical_focus_on = yes
	NOT = { has_war_with = ROOT }
	OR = {
		AND = {
			DAI_is_historical_axis_license_bloc_member = yes
			ROOT = { DAI_is_historical_axis_license_bloc_member = yes }
		}
		AND = {
			DAI_is_historical_allied_license_bloc_member = yes
			ROOT = { DAI_is_historical_allied_license_bloc_member = yes }
		}
	}
}

DAI_is_historical_license_partner_of_prev = {
	is_historical_focus_on = yes
	NOT = { has_war_with = PREV }
	OR = {
		AND = {
			DAI_is_historical_axis_license_bloc_member = yes
			PREV = { DAI_is_historical_axis_license_bloc_member = yes }
		}
		AND = {
			DAI_is_historical_allied_license_bloc_member = yes
			PREV = { DAI_is_historical_allied_license_bloc_member = yes }
		}
	}
}

DAI_is_license_partner_of_root = {
	OR = {
		is_in_faction_with = ROOT
		ROOT = { is_subject_of = PREV }
		is_subject_of = ROOT
		DAI_is_historical_license_partner_of_root = yes
	}
}

DAI_is_human_license_partner_of_root = {

	DAI_is_license_partner_of_root = yes
}

DAI_is_ai_license_partner_of_root = {

	DAI_is_license_partner_of_root = yes
}

DAI_has_ai_license_partner = {
	any_other_country = {
		DAI_is_ai_license_partner_of_root = yes
	}
}
