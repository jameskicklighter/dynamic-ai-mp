# These are live phase checks. Use flags instead only if we want a country to
# remember a chosen phase after the underlying map/date situation changes.
DAI_allies_france_has_fallen = {
	OR = {
		has_global_flag = FRA_already_surrendered
		FRA = {
			OR = {
				has_capitulated = yes
				exists = no
			}
		}
	}
}

DAI_allies_have_french_foothold = {
	OR = {
		14 = { is_controlled_by_ROOT_or_ally = yes } # Brittany
		15 = { is_controlled_by_ROOT_or_ally = yes } # Normandy
		16 = { is_controlled_by_ROOT_or_ally = yes } # Paris
		29 = { is_controlled_by_ROOT_or_ally = yes } # Pas-de-Calais
	}
}

}

DAI_allies_d_day_base_conditions = {
	is_historical_focus_on = yes
	OR = {
		original_tag = ENG
		is_in_faction_with = ENG
	}
	has_war_with = GER
	GER = { has_capitulated = no }
	DAI_allies_france_has_fallen = yes
}

DAI_allies_soviet_d_day_pressure = {
	country_exists = SOV
	SOV = {
		has_capitulated = no
		has_war_with = GER
		surrender_progress > 0.30
	}
}

DAI_allies_soviet_cannot_resist_germany = {
	OR = {
		NOT = { country_exists = SOV }
		SOV = {
			OR = {
				has_capitulated = yes
				is_subject_of = GER
				is_in_faction_with = GER
			}
		}
	}
}

DAI_allies_soviet_requires_early_d_day = {
	OR = {
		DAI_allies_soviet_cannot_resist_germany = yes
		AND = {
			country_exists = SOV
			SOV = {
				has_capitulated = no
				has_war_with = GER
				surrender_progress > 0.50
			}
		}
	}
}

DAI_allies_italy_is_in_chaos_for_d_day = {
	OR = {
		NOT = { has_war_with = ITA }
		ITA = { exists = no }
		ITA = { has_capitulated = yes }
		ITA = { has_civil_war = yes }
		ITA = { surrender_progress > 0.2 }
	}
}

DAI_allies_africa_italy_ready_for_normandy = {
	DAI_allies_north_africa_is_contested = no
	DAI_allies_italy_is_in_chaos_for_d_day = yes
}

# Severe Soviet pressure or collapse bypasses the calendar. The existing dated
# Soviet-pressure, Africa/Italy, and historical 1944 paths remain unchanged.
DAI_allies_normandy_can_build_up = {
	OR = {
		has_global_flag = DAI_allies_d_day_prep_started
		DAI_allies_soviet_requires_early_d_day = yes
		AND = {
			date > 1941.12.31 # Earliest date for the existing build-up paths: 1942.1.1
			OR = {
				date > 1944.1.1
				DAI_allies_soviet_d_day_pressure = yes
				AND = {
					date > 1943.1.1
					DAI_allies_africa_italy_ready_for_normandy = yes
				}
			}
		}
	}
}

DAI_allies_normandy_preparation_has_matured = {
	has_global_flag = {
		flag = DAI_allies_d_day_prep_started
		days > 180
	}
}

DAI_allies_normandy_date_gate_is_open = {
	OR = {
		date > 1944.6.1
		AND = {
			date > 1942.6.1
			has_global_flag = DAI_allies_d_day_prep_started
		}
		AND = {
			date > 1943.6.1
			DAI_allies_africa_italy_ready_for_normandy = yes
		}
	}
}

DAI_allies_normandy_initial_release_ready = {
	DAI_allies_focus_normandy = yes
	NOT = { has_global_flag = DAI_allies_d_day_initial_release_done }
	DAI_allies_normandy_preparation_has_matured = yes
	DAI_allies_normandy_date_gate_is_open = yes
}

DAI_allies_normandy_repeat_release_open = {
	DAI_allies_focus_normandy = yes
	has_global_flag = DAI_allies_d_day_initial_release_done
	has_global_flag = {
		flag = DAI_allies_d_day_release_cycle_started
		days < 90
	}
}

# Go-time is delayed after build-up, then opens on repeat cycle windows so Allied countries sync.
DAI_allies_normandy_can_go = {
	OR = {
		DAI_allies_normandy_initial_release_ready = yes
		DAI_allies_normandy_repeat_release_open = yes
	}
}
