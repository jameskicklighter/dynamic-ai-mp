# Compact Sheep-style D-Day layer: hold Normandy until conditions are right,
# stage forces in Britain, then force a focused landing and beachhead buildup.

DAI_allies_d_day_normandy_hardstop = {
	allowed = {
		is_historical_focus_on = yes
		OR = {
			original_tag = USA
			original_tag = ENG
		}
	}
	enable = {
		has_war_with = GER
		GER = { has_capitulated = no }
		FRA = {
			OR = {
				has_capitulated = yes
				exists = no
			}
		}
		NOT = {
			OR = {
				14 = { is_controlled_by_ROOT_or_ally = yes } # Brittany
				15 = { is_controlled_by_ROOT_or_ally = yes } # Normandy
				16 = { is_controlled_by_ROOT_or_ally = yes } # Paris
				29 = { is_controlled_by_ROOT_or_ally = yes } # Pas-de-Calais
			}
		}
		NOT = {
			OR = {
				AND = {
					date > 1942.6.1
					SOV = {
						has_war_with = GER
						surrender_progress > 0.25
					}
				}
				AND = {
					date > 1943.6.1
					OR = {
						date > 1944.1.1
						ITA = { has_capitulated = yes }
						ITA = { has_civil_war = yes }
						ITA = { surrender_progress > 0.2 }
					}
				}
			}
		}
	}
	abort_when_not_enabled = yes

	ai_strategy = {
		type = front_control
		area = normandy_landing_zone
		ratio = 0.0
		priority = 300
		ordertype = invasion
		execute_order = no
	}
}
