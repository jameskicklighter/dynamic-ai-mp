british_help_against_germany = {
	allowed = {
		original_tag = POL
	}
	enable = {
		OR = {
			GER = { has_country_flag = GER_second_vienna_award_flag }
			GER = { has_war_with = POL }
		}
		country_exists = ENG
	}
	abort = {
		NOT = { country_exists = ENG }
	}

	ai_strategy = {
		type = befriend
		id = "ENG"
		value = 200
	}
	ai_strategy = {
		type = alliance
		id = "ENG"
		value = 200
	}
}
