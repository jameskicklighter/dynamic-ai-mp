DAI_historical_axis_expeditionaries_to_GER = {
	allowed = {
		OR = {
			original_tag = BUL
			original_tag = HUN
			original_tag = ROM
		}
	}

	enable = {
		DAI_historical_axis_expeditionaries_to_GER_enable = yes
	}

	abort_when_not_enabled = yes

	# Historical Axis minors should offer Germany real expeditionary forces if the engine accepts this diplomatic-action desire.
	# Runtime debug logs for this condition are emitted by DAI_update_historical_axis_debug.
	ai_strategy = {
		type = diplo_action_desire
		id = GER
		target = send_expeditionary_force
		value = 10000
	}
}
