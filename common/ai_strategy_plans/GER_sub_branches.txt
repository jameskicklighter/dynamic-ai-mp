# Protective branch: chosen if the following conditions are met:
# If we are democratic:
# - At least one of Poland and Baltic nations are not fascist/communist.
# Else (non-aligned):
# - At least 2 of Poland and Baltic nations share our government.
GER_realpolitik_see_to_the_eastern_front = {
	name = "See to the Eastern Front sub-branch"
	desc = "desc"

	allowed = {
		original_tag = GER
		has_dlc = "Gotterdammerung"
	}
	enable = {
		has_completed_focus = GER_realpolitik
		IF = { limit = { has_government = democratic }
			OR = {
				EST = {
					exists = yes
					is_subject = no
					NOT = {
						is_in_faction_with = SOV
						has_war_with = ROOT
					}
					NOT = { has_government = fascism }
					NOT = { has_government = communism }
				}
				LAT = {
					exists = yes
					is_subject = no
					NOT = {
						is_in_faction_with = SOV
						has_war_with = ROOT
					}
					NOT = { has_government = fascism }
					NOT = { has_government = communism }
				}
				LIT = {
					exists = yes
					is_subject = no
					NOT = {
						is_in_faction_with = SOV
						has_war_with = ROOT
					}
					NOT = { has_government = fascism }
					NOT = { has_government = communism }
				}
				POL = {
					exists = yes
					is_subject = no
					NOT = {
						is_in_faction_with = SOV
						has_war_with = ROOT
					}
					NOT = { has_government = fascism }
					NOT = { has_government = communism }
				}
			}
		}
		# Can only be non-aligned.
		ELSE = {
			count_triggers = {
				amount = 2
				EST = {
					has_government = ROOT
					exists = yes
					is_subject = no
					NOT = {
						is_in_faction_with = SOV
						has_war_with = ROOT
					}
				}
				LAT = {
					has_government = ROOT
					exists = yes
					is_subject = no
					NOT = {
						is_in_faction_with = SOV
						has_war_with = ROOT
					}
				}
				LIT = {
					has_government = ROOT
					exists = yes
					is_subject = no
					NOT = {
						is_in_faction_with = SOV
						has_war_with = ROOT
					}
				}
				POL = {
					has_government = ROOT
					exists = yes
					is_subject = no
					NOT = {
						is_in_faction_with = SOV
						has_war_with = ROOT
					}
				}
			}
		}
	}
	abort = {

	}
