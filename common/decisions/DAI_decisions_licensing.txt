DAI_production_license_decisions = {

	DAI_share_mech_license_with_ai = {
		icon = GFX_decision_generic_research
		allowed = {
			always = yes
		}
		visible = {
			DAI_has_ai_license_partner = yes
			has_tech = mechanised_infantry
			has_country_flag = DAI_mechanized_cost_upgrade_done
			any_other_country = {
				DAI_is_ai_license_partner_of_root = yes
				NOT = { has_tech = mechanised_infantry }
				NOT = { has_country_flag = DAI_mech_licensed }
			}
		}
		available = {
			DAI_has_ai_license_partner = yes
			has_tech = mechanised_infantry
			has_country_flag = DAI_mechanized_cost_upgrade_done
		}
		cost = 0
		fire_only_once = no
		ai_will_do = { base = 999 }
		complete_effect = {
			every_other_country = {
				limit = {
					DAI_is_ai_license_partner_of_root = yes
					NOT = { has_tech = mechanised_infantry }
					NOT = { has_country_flag = DAI_mech_licensed }
				}
				DAI_set_license_relation_recipient_to_provider = yes
				ROOT = {
					create_production_license = {
						target = PREV
						equipment = { type = mechanized_equipment_1 }
						cost_factor = 0
					}
				}
				set_country_flag = DAI_mech_licensed
				if = {
					limit = { NOT = { has_tech = DAI_enable_mech } }
					set_technology = { DAI_enable_mech = 1 popup = no }
				}
			}
		}
	}
