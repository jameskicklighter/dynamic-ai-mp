
DAI_daily_update = {

	if = { limit = { is_ai = yes }

		# If we are a dynamic tag, we going with the flow instead.
		# So just use the normal tag.
		if = { limit = { tag = GER } DAI_set_build_order_GER = yes }
		if = { limit = { tag = FRA } DAI_set_build_order_FRA = yes }

	}
}

DAI_monthly_update = {

	if = { limit = { is_ai = yes }

		# Strategy flags
		DAI_update_allied_d_day_flags = yes

		# Laws
		DAI_set_desired_trade_law = yes

		# Military
		DAI_set_wants_armor_flag = yes
		DAI_set_armor_template_flags = yes

	}
}

DAI_update_allied_d_day_flags = {

	if = {
		limit = {
			is_historical_focus_on = yes
			date > 1942.1.1
			GER = { has_capitulated = no }
			SOV = {
				has_war_with = GER
				surrender_progress > 0.30
			}
		}
		set_global_flag = DAI_allies_soviet_d_day_pressure
	}

	if = {
		limit = {
			has_global_flag = DAI_allies_soviet_d_day_pressure
			OR = {
				NOT = { is_historical_focus_on = yes }
				NOT = { country_exists = GER }
				GER = { has_capitulated = yes }
				NOT = { country_exists = SOV }
				SOV = {
					OR = {
						NOT = { has_war_with = GER }
						surrender_progress < 0.15
					}
				}
			}
		}
		clr_global_flag = DAI_allies_soviet_d_day_pressure
	}

}
