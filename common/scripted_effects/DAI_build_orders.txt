DAI_set_build_order_ENG = {

	set_variable = { DAI_build_order_wants = -1 }

	IF = { limit = { DAI_ENG_is_historical_ai = yes }

		IF = {
			limit = {
				NOT = { has_idea_with_trait = silent_workhorse }
				has_available_idea_with_traits = { idea = silent_workhorse limit = 1 }
			}
			set_variable = { DAI_build_order_wants = token:ENG_philip_kerr }
		}
		ELSE_IF = {
			limit = {
				DAI_build_order_should_take_partial_mob = yes
			}
			set_variable = { DAI_build_order_wants = token:partial_economic_mobilisation }
		}
		ELSE_IF = {
			limit = {
				NOT = { has_idea_with_trait = army_chief_offensive_2 }
				has_available_idea_with_traits = { idea = army_chief_offensive_2 limit = 1 }
			}
			set_variable = { DAI_build_order_wants = token:ENG_john_vereker }
		}
		ELSE_IF = {
			limit = {
				NOT = { has_idea_with_trait = air_chief_reform_2 }
				has_available_idea_with_traits = { idea = air_chief_reform_2 limit = 1 }
			}
			set_variable = { DAI_build_order_wants = token:ENG_cyril_newall }
		}
		ELSE_IF = {
			limit = {
				NOT = { has_idea_with_trait = navy_chief_maneuver_2 }
				has_available_idea_with_traits = { idea = navy_chief_maneuver_2 limit = 1 }
			}
			set_variable = { DAI_build_order_wants = token:ENG_ernle_chatfield }
		}
		ELSE_IF = {
			limit = {
				NOT = { has_idea_with_trait = war_industrialist }
				has_available_idea_with_traits = { idea = war_industrialist limit = 1 }
			}
			set_variable = { DAI_build_order_wants = token:ENG_maxwell_aitken }
		}
		ELSE_IF = {
			limit = {
				NOT = { has_idea_with_trait = grand_battle_plan_expert }
				has_available_idea_with_traits = { idea = grand_battle_plan_expert limit = 1 }
			}
			set_variable = { DAI_build_order_wants = token:ENG_bernard_montgomery }
		}

		log = "[GetYear] [GetMonth] | AI | [Root.GetName] | DAI_build_order_wants: = [?DAI_build_order_wants]"

	}

}

DAI_set_build_order_GER = {

	set_variable = { DAI_build_order_wants = -1 }

	IF = { limit = { DAI_GER_is_historical_ai = yes }

		IF = {
			limit = {
				has_dynamic_modifier = { modifier = GER_mefo_bills_modifier }
				GER_hjalmar_schacht = { is_hired_as_advisor = no }
				has_available_idea_with_traits = { idea = GER_mefo_bills_architect limit = 1 }
			}
			set_variable = { DAI_build_order_wants = token:GER_hjalmar_schacht }
		}
		ELSE_IF = {
			limit = {
				# We are only concerned about early build order. Don't do an extensive check for conditions,
				# as they should normally be met by the logic of the build order.
				NOT = { has_idea = war_economy }
				NOT = { has_idea = tot_economic_mobilisation }
				date < 1937.1.1
				has_war_support > 0.5
			}
			set_variable = { DAI_build_order_wants = token:war_economy }
		}
		ELSE_IF = {
			limit = {
				NOT = { has_idea_with_trait = GER_reich_marshal }
				has_available_idea_with_traits = { idea = GER_reich_marshal limit = 1 }
			}
			set_variable = { DAI_build_order_wants = token:GER_hermann_goring }
		}
		ELSE_IF = {
			limit = {
				NOT = { has_idea_with_trait = army_chief_organizational_2 }
				has_available_idea_with_traits = { idea = army_chief_organizational_2 limit = 1 }
			}
			set_variable = { DAI_build_order_wants = token:GER_ludwig_beck }
		}
		ELSE_IF = {
			limit = {
				NOT = { has_idea = GER_organization_todt }
				NOT = { has_idea = GER_organization_todt_improved }
				has_completed_focus = GER_form_organization_todt
			}
			set_variable = { DAI_build_order_wants = token:GER_organization_todt }
		}
		ELSE_IF = {
			limit = {
				NOT = { has_idea_with_trait = navy_chief_decisive_battle_2 }
				has_available_idea_with_traits = { idea = navy_chief_decisive_battle_2 limit = 1 }
			}
			set_variable = { DAI_build_order_wants = token:GER_erich_raeder }
		}
		# ELSE_IF = {
		# 	limit = {
		# 		NOT = { has_idea_with_trait = navy_chief_decisive_battle_2 }
		# 		has_available_idea_with_traits = { idea = navy_chief_decisive_battle_2 limit = 1 }
		# 	}
		# 	set_variable = { DAI_build_order_wants = token:GER_erich_raeder }
		# }

		log = "[GetYear] [GetMonth] | AI | [Root.GetName] | DAI_build_order_wants: = [?DAI_build_order_wants]"

	}

}
