DAI_set_license_relation_recipient_to_provider = {
	if = {
		limit = {
			NOT = { has_country_flag = DAI_license_relation_from_@ROOT }
		}
		set_country_flag = DAI_license_relation_from_@ROOT
		diplomatic_relation = {
			country = ROOT
			relation = licence_production_access
			active = yes
		}
	}
}

DAI_notify_human_license_opportunities_from_provider = {
	every_other_country = {
		limit = {
			DAI_is_human_license_partner_of_root = yes
			DAI_has_license_notification_opportunity_from_root = yes
		}
		country_event = { id = DAI_licensing.1 }
		DAI_mark_license_notifications_from_provider = yes
	}
}
